  23. /**
  24.  * Creates and registers a StateMachine described in XML.
  25.  * 
  26.  * This allows reconfiguration of the StateMachine
  27.  * without changing any code, as well as making it
  28.  * easier than creating all the <b>State</b>
  29.  * instances and registering them with the
  30.  * <b>StateMachine</b> at startup time.
  31.  * 
  32.  * One way to acheive this setup is to use a command controller that
  33.  * could be added a application startup.  Such a controller could look
  34.  * like this:
  35.  * <code>
  36.  * //From the PureMVC AS3 StopWatch Demo
  37.  * class InjectFSMCommand extends SimpleCommand
  38.  * {
  39.  *     public function execute ( INotification $notification )
  40.  *     {
  41.  *         // Create the FSM definition
  42.  *         $fsmStr = <<<XML
  43.  *         <fsm initial="StopWatch/states/ready">
  44.  *             <state name="StopWatch/states/ready" entering="resetDisplay">
  45.  *                 <transition action="StopWatch/actions/start" target="StopWatch/states/running"/>
  46.  *             </state>
  47.  *             <state name="StopWatch/states/running" entering="ensureTimer">
  48.  *                 <transition action="StopWatch/actions/split" target="StopWatch/states/paused"/>
  49.  *                 <transition action="StopWatch/actions/stop" target="StopWatch/states/stopped"/>
  50.  *             </state>
  51.  *             <state name="StopWatch/states/paused" entering="freezeDisplay">
  52.  *                 <transition action="StopWatch/actions/unsplit" target="StopWatch/states/running"/>
  53.  *                 <transition action="StopWatch/actions/stop" target="StopWatch/states/stopped"/>
  54.  *             </state>
  55.  *             <state name="StopWatch/states/stopped" entering="stopTimer">
  56.  *                 <transition action="StopWatch/actions/reset" target="StopWatch/states/ready"/>
  57.  *             </state>
  58.  *         </fsm>XML;
  59.  * 
  60.  *         $fsm = new SimpleXMLElement($fsmStr);
  61.  * 
  62.  *         $injector = new FSMInjector($fsm);
  63.  *         $injector->initializeNotifier('FSMInjectorTest');
  64.  *         $injector->inject();
  65.  *     }
  66.  * }
  67.  * </code>
  68.  * 
  69.  * @see State
  70.          org/puremvc/php/multicore/utilities/statemachine/State.php
  71.  * @see StateMachine
  72.          org/puremvc/php/multicore/utilities/statemachine/StateMachine.php
  73.  * 
  74.  * @package org.puremvc.php.multicore.utilities.statemachine
  75.  */
  76. class FSMInjector extends Notifier
  77. {
  78.     /**
  79.      * The XML FSM definition
  80.      * @var SimpleXMLElement 
  81.      */
  82.     protected $fsm;
  83.     
  84.     /**
  85.      * The List of State objects
  86.      * @var array 
  87.      */
  88.     protected $stateList = null;
  89.     
  90.     /**
  91.      * Constructor.
  92.      * 
  93.      * @param SimpleXMLElement $fsm The XML FSM definition
  94.      * @return FSMInjector 
  95.      */
  96.     public function __construct( SimpleXMLElement $fsm ) 
  97.     {
  98.         $this->fsm = $fsm;
  99.     }
  100.     
  101.     /**
  102.      * Inject the <b>StateMachine</b> into the PureMVC apparatus.
  103.      * 
  104.      * Creates the <b>StateMachine</b> instance, registers all the states
  105.      * and registers the <b>StateMachine</b> with the <b>IFacade</b>.
  106.      * 
  107.      * @return void 
  108.      */
  109.     public function inject()
  110.     {
  111.         // Create the StateMachine
  112.         $stateMachine = new StateMachine();
  113.         
  114.         // Register all the states with the StateMachine
  115.         foreach ( $this->getStates() as $state )
  116.         { 
  117.             $stateMachine->registerState( $state, $this->isInitial( $state->name ) );
  118.         }                
  119.         // Register the StateMachine with the facade
  120.         $this->facade()->registerMediator( $stateMachine );
  121.     }
  122.  
  123.     
  124.     /**
  125.      * Get the state definitions.
  126.      * 
  127.      * Creates and returns the array of State objects
  128.      * from the FSM on first call, subsequently returns
  129.      * the existing array.
  130.      * 
  131.      * @return array 
  132.      */
  133.     protected function getStates()
  134.     {
  135.         if (!isset($this->stateList)) 
  136.         {
  137.             $this->stateList = array();
  138.             
  139.             //$stateDefs = $this->fsm->state;
  140.             
  141.             foreach($this->fsm->state as $stateDef)
  142.             {
  143.                 $state = $this->createState( $stateDef );
  144.                 array_push($this->stateList, $state);
  145.             }
  146.         } 
  147.         return $this->stateList;
  148.     }
  149.  
  150.     /**
  151.      * Creates a <b>State</b> instance from its XML definition.
  152.      * 
  153.      * @param SimpleXMLElement $stateDef 
  154.      * @return State 
  155.       */
  156.     protected function createState( SimpleXMLElement $stateDef )
  157.     {
  158.         // Create State object
  159.         $name = (string)$stateDef['name'];
  160.         $entering = (string)$stateDef['entering'];
  161.         $exiting = (string)$stateDef['exiting'];
  162.         $changed = (string)$stateDef['changed'];
  163.         $state = new State( $name, $entering, $exiting, $changed );
  164.         
  165.         // Create transitions
  166.         foreach($stateDef->transition as $transDef)
  167.         {
  168.             $state->defineTrans( (string)$transDef['action'], (string)$transDef['target'] );
  169.         }
  170.  
  171.         return $state;
  172.     }
  173.  
  174.     /**
  175.      * Is the given state the initial state?
  176.      * 
  177.      * @param string $stateName 
  178.      * @return bool 
  179.      */
  180.     protected function isInitial( $stateName )
  181.     {
  182.         $initial = (string)$this->fsm['initial'];
  183.         return ($stateName == $initial);
  184.     }
  185.     
  186. }
