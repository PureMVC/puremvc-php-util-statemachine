  21.  
  22. /**
  23.  * A Finite State Machine implimentation.
  24.  * 
  25.  * Handles regisistration and removal of state definitions,
  26.  * which include optional entry and exit commands for each
  27.  * state.
  28.  * 
  29.  * @see Mediator
  30.          org/puremvc/php/multicore/patterns/mediator/Mediator.php          
  31.  * 
  32.  * @package org.puremvc.php.multicore.utilities.statemachine
  33.  */
  34. class StateMachine extends Mediator
  35. {
  36.     const NAME = "StateMachine";
  37.  
  38.     /**
  39.      * Action Notification name.
  40.      */ 
  41.     const ACTION = "StateMachine/notes/action";
  42.  
  43.     /**
  44.      *  Changed Notification name
  45.      */ 
  46.     const CHANGED = "StateMachine/notes/changed";
  47.     
  48.     /**
  49.      *  Cancel Notification name
  50.      */ 
  51.     const CANCEL = "StateMachine/notes/cancel";
  52.     
  53.     /**
  54.      * Map of States objects by name.
  55.      * @var array 
  56.      */
  57.     protected $states = array();
  58.     
  59.     /**
  60.      * The initial state of the FSM.
  61.      * @var State 
  62.      */
  63.     protected $initial;
  64.     
  65.     /**
  66.      * The transition has been canceled.
  67.      * @var bool 
  68.      */
  69.     protected $canceled;
  70.  
  71.     /**
  72.      * Constructor.
  73.      * 
  74.      * @return StateMachine 
  75.      */
  76.     public function __construct( )
  77.     {
  78.         parent::__construct( StateMachine::NAME );
  79.     }
  80.     
  81.     /**
  82.      * onRegister event
  83.      *
  84.      * Called by the <b>View</b> when the <b>Mediator</b> is registered.
  85.      *
  86.      * @return void 
  87.      */
  88.     public function onRegister()
  89.     {
  90.         if ( isset($this->initial) )
  91.         {
  92.             $this->transitionTo( $this->initial, null );
  93.         }    
  94.     }
  95.     
  96.     /**
  97.      * Registers the entry and exit commands for a given state.
  98.      * 
  99.      * @param State $state the state to which to register the above commands
  100.      * @param bool $initial boolean telling if this is the initial state of the system
  101.      * @return void 
  102.      */
  103.     public function registerState( State $state, $initial=false )
  104.     {
  105.         if ( is_null($state) || isset($this->states[ $state->name ]) ) return;
  106.         $this->states[ $state->name ] = $state;
  107.         if ( $initial ) $this->initial = $state; 
  108.     }
  109.     
  110.     /**
  111.      * Remove a state mapping.
  112.      * 
  113.      * Removes the entry and exit commands for a given state
  114.      * as well as the state mapping itself.
  115.      * 
  116.      * @param  string stateName
  117.      * @return void 
  118.      */
  119.     public function removeState( $stateName )
  120.     {
  121.         unset($this->states[ $stateName ]);
  122.     }
  123.     
  124.     /**
  125.      * Transitions to the given state from the current state.
  126.      * 
  127.      * Sends the <b>exiting</b> notification for the current state
  128.      * followed by the <b>entering</b> notification for the new state.
  129.      * Once finally transitioned to the new state, the <b>changed</b>
  130.      * notification for the new state is sent.
  131.      * 
  132.      * If a data parameter is provided, it is included as the body of all
  133.      * three state-specific transition notes.
  134.      * 
  135.      * Finally, when all the state-specific transition notes have been
  136.      * sent, a <b>StateMachine.CHANGED</b> note is sent, with the
  137.      * new <b>State</b> object as the <b>body</b> and the name of the
  138.      * new state in the <b>type</b>.
  139.      * 
  140.      * @param State $nextState the next State to transition to.
  141.      * @param object data is the optional Object that was sent in the <b>StateMachine.ACTION</b> notification body
  142.      * @return void 
  143.      */
  144.     protected function transitionTo( State $nextState, $data=null )
  145.     {
  146.         // Going nowhere?
  147.         if ( $nextState == null ) return;
  148.         
  149.         // Clear the cancel flag
  150.         $this->canceled = false;
  151.             
  152.         // Exit the current State 
  153.         if ( $this->getCurrentState() && $this->getCurrentState()->exiting ) $this->sendNotification( $this->getCurrentState()->exiting, $data, $nextState->name );
  154.         
  155.         // Check to see whether the exiting guard has canceled the transition
  156.         if ( $this->canceled ) 
  157.         {
  158.             $this->canceled = false;
  159.             return;
  160.         }
  161.         
  162.         // Enter the next State 
  163.         if ( $nextState->entering ) $this->sendNotification( $nextState->entering, $data );
  164.         
  165.         
  166.         // Check to see whether the entering guard has canceled the transition
  167.         if ( $this->canceled ) 
  168.         {
  169.             $this->canceled = false;
  170.             return;
  171.         }
  172.         
  173.         // change the current state only when both guards have been passed
  174.         $this->setCurrentState($nextState);
  175.         
  176.         // Send the notification configured to be sent when this specific state becomes current 
  177.         if ( $nextState->changed ) $this->sendNotification( $this->getCurrentState()->changed, $data );
  178.  
  179.         // Notify the app generally that the state changed and what the new state is 
  180.         $this->sendNotification( StateMachine::CHANGED, $this->getCurrentState(), $this->getCurrentState()->name );
  181.     
  182.     }
  183.     
  184.     /**
  185.      * Notification interests for the StateMachine.
  186.      */
  187.     public function listNotificationInterests()
  188.     {
  189.         return array( StateMachine::ACTION,
  190.                       StateMachine::CANCEL);
  191.     }
  192.     
  193.     /**
  194.      * Handle notifications the <b>StateMachine</b> is interested in.
  195.      * 
  196.      * <b>StateMachine::ACTION</b>: Triggers the transition to a new state.<BR>
  197.      * <b>StateMachine::CANCEL</b>: Cancels the transition if sent in response to the exiting note for the current state.<BR>
  198.      * 
  199.      * @param INotification $notification The <b>INotification</b> to be handled.
  200.      * @return void 
  201.      */
  202.     public function handleNotification( INotification $notification )
  203.     {
  204.         switch( $notification->getName() )
  205.         {
  206.             case StateMachine::ACTION:
  207.                 $action = $notification->getType();
  208.                 $target = $this->getCurrentState()->getTarget( $action );
  209.                 $newState = $this->states[ $target ];
  210.                 if ( $newState ) $this->transitionTo( $newState, $notification->getBody() );  
  211.                 break;
  212.                 
  213.             case StateMachine::CANCEL:
  214.                 $this->canceled = true;
  215.                 break;
  216.         }
  217.     }
  218.     
  219.     /**
  220.      * Get the current state.
  221.      *  
  222.      * @return State A State defining the machine's current state
  223.      */
  224.     protected function getCurrentState()
  225.     {
  226.         return $this->viewComponent;
  227.     }
  228.  
  229.     /**
  230.      * Set the current state.
  231.      * 
  232.      * @param State $state The current state to use
  233.      */
  234.     protected function setCurrentState( State $state )
  235.     {
  236.         $this->viewComponent = $state;
  237.     }
  238.     
  239. }
